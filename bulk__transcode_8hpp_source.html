55template <codec C, legacy_byte_range R>
56std::vector<char32_t> decode_to(R&& source) {
57 std::vector<char32_t> result;
58 detail::reserve_if_sized(result, source);
59
60 if constexpr (detail::bulk_single_byte_codec<C>) {
61 const auto* table = detail::get_decode_table<C>();
62 for (auto byte : source)
63 result.push_back(detail::decode_single_byte_at(detail::to_unsigned_byte(byte), table));
64 } else {
65 for (char32_t code_point : std::forward<R>(source) | whatwg_decode<C>)
66 result.push_back(code_point);
67 }
68
69 return result;
