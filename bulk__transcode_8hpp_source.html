3// Convenience helpers: decode_to, encode_to, decode_into, encode_into.
4//
5// These are thin wrappers over `view | ranges::to<Container>()` and
6// `ranges::copy(view, output)`. They are NOT proposed for standardization
7// because the standard library already provides the necessary composition
8// tools (ranges::to, ranges::copy) and the views compose naturally with
9// them. This header exists for practical convenience in codebases that
10// prefer a named-function call style.

40template <codec C, legacy_byte_range R>
41std::vector<char32_t> decode_to(R&& source) {
42 return std::forward<R>(source) | whatwg_decode<C> | std::ranges::to<std::vector<char32_t>>();
43}
44
45template <codec C, typename Container = std::string, unicode_scalar_range R>
46Container encode_to(R&& source) {
47 return std::forward<R>(source) | whatwg_encode<C> | std::ranges::to<Container>();
48}

50template <codec C, legacy_byte_range R, std::output_iterator<char32_t> Output>
51constexpr void decode_into(R&& source, Output output) {
52 std::ranges::copy(std::forward<R>(source) | whatwg_decode<C>, output);
53}
