52// Binary search the ranges table to convert a linear 4-byte pointer to a codepoint.
53// Returns U+FFFD if the pointer falls in a gap between ranges or exceeds U+10FFFF.
54// WHATWG special cases: pointer 7457 → U+E7C7; gaps above pointer 39419.
55constexpr char32_t gb18030_ranges_decode(std::uint32_t pointer) {
56 if ((pointer > 39419 && pointer < 189000) || pointer > 1237575)
57 return U'\xFFFD';
58 if (pointer == 7457)
59 return char32_t(0xE7C7);
60 const auto* table = tables::gb18030_ranges;
61 int n = tables::gb18030_ranges_count;
62 int lo = 0, hi = n - 1;
63 while (lo <= hi) {
64 int mid = lo + (hi - lo) / 2;
65 if (mid + 1 < n && table[mid + 1].pointer <= pointer) {
66 lo = mid + 1;
67 } else if (table[mid].pointer > pointer) {
68 hi = mid - 1;
69 } else {
70 auto offset = pointer - table[mid].pointer;
71 auto cp = table[mid].codepoint + offset;
72 if (mid + 1 < n && cp >= table[mid + 1].codepoint)
73 std::unreachable();
74 if (cp > 0x10FFFF)
75 std::unreachable();
76 return static_cast<char32_t>(cp);
77 }
78 }
79 std::unreachable();
80}

82// Binary search the ranges table to convert a codepoint to a linear 4-byte pointer.
83// WHATWG special case: U+E7C7 → pointer 7457.
84constexpr std::uint32_t gb18030_ranges_encode(char32_t cp) {
85 if (cp == char32_t(0xE7C7))
86 return 7457;
87 const auto* table = tables::gb18030_ranges;
88 int n = tables::gb18030_ranges_count;
89 int lo = 0, hi = n - 1;
90 while (lo <= hi) {
91 int mid = lo + (hi - lo) / 2;
92 if (mid + 1 < n && table[mid + 1].codepoint <= cp) {
93 lo = mid + 1;
94 } else if (table[mid].codepoint > cp) {
95 hi = mid - 1;
96 } else {
97 return table[mid].pointer + (cp - table[mid].codepoint);
98 }
99 }
100 std::unreachable();
101}

103template <std::input_iterator I, std::sentinel_for<I> S>
104constexpr gb18030_decode_result gb18030_decode_one(I& current, S end) {
105 if (current == end)
106 return {{}, whatwg_error::truncated_sequence, true};
107
108 auto lead = static_cast<unsigned char>(*current);
109 ++current;
110
111 if (lead < 0x80)
112 return {static_cast<char32_t>(lead), {}, false};
113
114 if (lead == 0x80)
115 return {U'\x20AC', {}, false};
116
117 if (lead < 0x81 || lead > 0xFE)
118 return {{}, whatwg_error::invalid_byte, true};
119
120 if (current == end)
121 return {{}, whatwg_error::truncated_sequence, true};
122
123 auto second = static_cast<unsigned char>(*current);
124
125 // 4-byte GB18030: second byte is a decimal digit 0x30–0x39.
126 if (second >= 0x30 && second <= 0x39) {
127 ++current; // consume second
128 if (current == end)
129 return {{}, whatwg_error::truncated_sequence, true};
130 auto third = static_cast<unsigned char>(*current);
131 if (third < 0x81 || third > 0xFE) {
132 // WHATWG: prepend second to ioQueue. Third is not consumed.
133 gb18030_decode_result err{{}, whatwg_error::invalid_byte, true, {}, 0};
134 err.replay[0] = second;
135 err.replay_count = 1;
136 return err;
137 }
138 ++current; // consume third
139 if (current == end)
140 return {{}, whatwg_error::truncated_sequence, true};
141 auto fourth = static_cast<unsigned char>(*current);
142 if (fourth < 0x30 || fourth > 0x39) {
143 // WHATWG: prepend second, third to ioQueue. Fourth is not consumed.
144 gb18030_decode_result err{{}, whatwg_error::invalid_byte, true, {}, 0};
145 err.replay[0] = second;
146 err.replay[1] = third;
147 err.replay_count = 2;
148 return err;
149 }
150 ++current; // consume fourth
151
152 std::uint32_t pointer =
153 ((lead - 0x81u) * 10u + (second - 0x30u)) * 126u * 10u + (third - 0x81u) * 10u + (fourth - 0x30u);
154 char32_t cp = gb18030_ranges_decode(pointer);
155 if (cp == U'\xFFFD')
156 return {{}, whatwg_error::invalid_byte, true};
157 return {cp, {}, false};
158 }
159
160 // 2-byte GBK: trail byte 0x40–0xFE, not 0x7F.
161 if (second < 0x40 || second > 0xFE || second == 0x7F) {
162 // WHATWG: if byte < 0x80, prepend (don't consume); otherwise consume.
163 if (second >= 0x80)
164 ++current;
165 return {{}, whatwg_error::invalid_byte, true};
166 }
167
168 ++current; // consume second
169
170 int offset = second > 0x7F ? 1 : 0;
171 int index = (lead - 0x81) * 190 + (second - 0x40) - offset;
172
173 char32_t cp = tables::gbk[index];
174 if (cp == 0)
175 std::unreachable();
176
177 return {cp, {}, false};
178}

180constexpr gb18030_encode_result gb18030_encode_one(char32_t cp) {
181 gb18030_encode_result r{};
182
183 // ASCII passthrough
184 if (cp < 0x80) {
185 r.bytes[0] = static_cast<unsigned char>(cp);
186 r.count = 1;
187 return r;
188 }
189
190 // Search GBK 2-byte table
191 for (int i = 0; i < 23940; ++i) {
192 if (tables::gbk[i] == cp) {
193 int lead = i / 190 + 0x81;
194 int trail = i % 190 + 0x40;
195 if (trail >= 0x7F)
196 ++trail;
197 r.bytes[0] = static_cast<unsigned char>(lead);
198 r.bytes[1] = static_cast<unsigned char>(trail);
199 r.count = 2;
200 return r;
201 }
202 }
203
204 // Encode via GB18030 4-byte ranges
205 std::uint32_t pointer = gb18030_ranges_encode(cp);
206 std::uint32_t p = pointer;
207 unsigned char b4 = static_cast<unsigned char>(p % 10 + 0x30);
208 p /= 10;
209 unsigned char b3 = static_cast<unsigned char>(p % 126 + 0x81);
210 p /= 126;
211 unsigned char b2 = static_cast<unsigned char>(p % 10 + 0x30);
212 p /= 10;
213 unsigned char b1 = static_cast<unsigned char>(p + 0x81);
214 r.bytes[0] = b1;
215 r.bytes[1] = b2;
216 r.bytes[2] = b3;
217 r.bytes[3] = b4;
218 r.count = 4;
219 return r;
220}
