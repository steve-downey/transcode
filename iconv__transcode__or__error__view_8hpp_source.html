20// iconv_transcode_or_error_view<IconvFns, R>
21//
22// Like iconv_transcode_view but yields std::expected<char, iconv_error> so
23// that EILSEQ (invalid sequence), EINVAL-at-end (incomplete sequence), and
24// E2BIG-with-nothing-written (output buffer too small) are surfaced to the
25// caller instead of being silently discarded.
26//
27// The iterator is move-only for the same reason as iconv_transcode_view:
28// iconv_t is an OS-managed handle that cannot be copied.
29template <typename IconvFns, std::ranges::input_range R>
30 requires legacy_byte_range<R>
31class iconv_transcode_or_error_view : public std::ranges::view_interface<iconv_transcode_or_error_view<IconvFns, R>> {
32 R base_;
33 IconvFns fns_;
34 const char* from_;
35 const char* to_;
36 std::span<char> buffer_;
37
38 public:
39 class iterator {
40 using base_iter = std::ranges::iterator_t<R>;
41 using base_sent = std::ranges::sentinel_t<R>;
42 using result_t = std::expected<char, iconv_error>;
43
44 iconv_t handle_;
45 IconvFns fns_;
46 std::span<char> buffer_;
47 char* output_pos_;
48 char* output_end_;
49 char staging_[4];
50 size_t staging_len_;
51 base_iter current_;
52 base_sent end_;
53 bool done_;
54 bool has_error_;
55 iconv_error error_value_;
56
57 iterator(const iterator&) = delete;
58 iterator& operator=(const iterator&) = delete;
59
60 void load();
61
62 friend class iconv_transcode_or_error_view;
63 iterator(iconv_t handle, IconvFns fns, std::span<char> buffer, base_iter current, base_sent end);
64
65 public:
66 using iterator_concept = std::input_iterator_tag;
67 using value_type = result_t;
68 using difference_type = std::ptrdiff_t;
69 using reference = result_t;
70
71 iterator(iterator&&) noexcept;
72 iterator& operator=(iterator&&) noexcept;
73 ~iterator();
74
75 result_t operator*() const;
76 iterator& operator++();
77 void operator++(int);
78
79 friend bool operator==(const iterator& it, std::default_sentinel_t) { return it.done_; }
80 };

82 public:
83 explicit iconv_transcode_or_error_view(
84 R base, IconvFns fns, const char* from, const char* to, std::span<char> buf);

88};

135template <typename IconvFns, std::ranges::input_range R>
136 requires legacy_byte_range<R>
137void iconv_transcode_or_error_view<IconvFns, R>::iterator::load() {
138 while (true) {
139 if (current_ != end_) {
140 staging_[staging_len_++] = static_cast<char>(*current_);
141 ++current_;
142 } else if (staging_len_ == 0) {
143 done_ = true;
144 return;
145 }
146
147 char* in_ptr = staging_;
148 size_t inleft = staging_len_;
149 char* out_ptr = buffer_.data();
150 size_t outleft = buffer_.size();
151 size_t rc = fns_.convert(handle_, &in_ptr, &inleft, &out_ptr, &outleft);
152
153 output_pos_ = buffer_.data();
154 output_end_ = out_ptr;
155
156 size_t consumed = staging_len_ - inleft;
157 if (consumed > 0) {
158 for (size_t i = 0; i < inleft; ++i)
159 staging_[i] = staging_[consumed + i];
160 staging_len_ = inleft;
161 }
162
163 if (rc != (size_t)-1) {
164 if (output_pos_ < output_end_)
165 return;
166 if (staging_len_ == 0 && current_ == end_) {
167 done_ = true;
168 return;
169 }
170 continue;
171 }
172
173 if (errno == EINVAL) {
174 // Yield any output produced before the incomplete sequence.
175 if (output_pos_ < output_end_)
176 return;
177 if (current_ == end_) {
178 // Incomplete trailing sequence — report error; next load() sets done_.
179 staging_len_ = 0;
180 has_error_ = true;
181 error_value_ = iconv_error::incomplete_sequence;
182 return;
183 }
184 // More input available — accumulate another byte.
185 continue;
186 }
187
188 if (errno == EILSEQ) {
189 // Yield any output produced before the invalid byte.
190 if (output_pos_ < output_end_)
191 return;
192 // Skip one bad staging byte and report the error.
193 if (staging_len_ > 0) {
194 for (size_t i = 0; i + 1 < staging_len_; ++i)
195 staging_[i] = staging_[i + 1];
196 --staging_len_;
197 }
198 has_error_ = true;
199 error_value_ = iconv_error::invalid_sequence;
200 return;
201 }
202
203 // E2BIG: yield whatever was written.
204 if (output_pos_ < output_end_)
205 return;
206 // Nothing written; output buffer too small to hold even one unit.
207 // Skip one staging byte to avoid stalling, and report the error.
208 if (staging_len_ > 0) {
209 for (size_t i = 0; i + 1 < staging_len_; ++i)
210 staging_[i] = staging_[i + 1];
211 --staging_len_;
212 }
213 has_error_ = true;
214 error_value_ = iconv_error::output_full;
215 return;
216 }
217}

241template <typename IconvFns, std::ranges::input_range R>
242 requires legacy_byte_range<R>
243auto iconv_transcode_or_error_view<IconvFns, R>::iterator::operator=(iterator&& other) noexcept -> iterator& {
244 if (this != &other) {
245 if (handle_ != (iconv_t)-1)
246 fns_.close(handle_);
247 handle_ = other.handle_;
248 fns_ = std::move(other.fns_);
249 buffer_ = other.buffer_;
250 output_pos_ = other.output_pos_;
251 output_end_ = other.output_end_;
252 staging_len_ = other.staging_len_;
253 for (size_t i = 0; i < other.staging_len_; ++i)
254 staging_[i] = other.staging_[i];
255 current_ = std::move(other.current_);
256 end_ = std::move(other.end_);
257 done_ = other.done_;
258 has_error_ = other.has_error_;
259 error_value_ = other.error_value_;
260 other.handle_ = (iconv_t)-1;
261 other.done_ = true;
262 other.staging_len_ = 0;
263 other.has_error_ = false;
264 }
265 return *this;
266}
