30// Decode one EUC-JP character from the range [current, end).
31// ASCII bytes (0x00-0x7F) pass through directly (1 byte consumed).
32// 0x8E (SS2) is followed by a half-width katakana byte (0xA1-0xDF), 2 bytes total.
33// 0x8F (SS3) is followed by two JIS X 0212 bytes (0xA1-0xFE each), 3 bytes total.
34// 0xA1-0xFE (JIS X 0208 lead) is followed by a trail byte (0xA1-0xFE), 2 bytes total.
35// Advances current by the number of bytes consumed.
36template <std::input_iterator I, std::sentinel_for<I> S>
37constexpr euc_jp_decode_result euc_jp_decode_one(I& current, S end);
38
39// Encode one Unicode codepoint as EUC-JP.
40// ASCII (cp < 0x80) outputs 1 byte. U+00A5 maps to 0x5C; U+203E to 0x7E.
41// U+FF61-U+FF9F (half-width katakana) encode as 0x8E + 1 byte.
42// JIS X 0208 codepoints encode as 2 bytes (0xA1-0xFE range).
43// JIS X 0212 codepoints encode as 3 bytes (0x8F + 0xA1-0xFE + 0xA1-0xFE).
44// Returns is_error=true if the codepoint is not representable in EUC-JP.
45constexpr euc_jp_encode_result euc_jp_encode_one(char32_t cp);

51template <std::input_iterator I, std::sentinel_for<I> S>
52constexpr euc_jp_decode_result euc_jp_decode_one(I& current, S end) {
53 if (current == end)
54 return {{}, whatwg_error::truncated_sequence, true};
55
56 auto lead = static_cast<unsigned char>(*current);
57 ++current;
58
59 // ASCII passthrough
60 if (lead < 0x80)
61 return {static_cast<char32_t>(lead), {}, false};
62
63 // SS2 prefix: half-width katakana
64 if (lead == 0x8E) {
65 if (current == end)
66 return {{}, whatwg_error::truncated_sequence, true};
67 auto byte2 = static_cast<unsigned char>(*current);
68 ++current;
69 if (byte2 < 0xA1 || byte2 > 0xDF)
70 return {{}, whatwg_error::invalid_byte, true};
71 return {static_cast<char32_t>(0xFF61 + (byte2 - 0xA1)), {}, false};
72 }
73
74 // SS3 prefix: JIS X 0212 (3-byte sequence)
75 if (lead == 0x8F) {
76 if (current == end)
77 return {{}, whatwg_error::truncated_sequence, true};
78 auto b1 = static_cast<unsigned char>(*current);
79 ++current;
80 if (b1 < 0xA1 || b1 > 0xFE)
81 return {{}, whatwg_error::invalid_byte, true};
82 if (current == end)
83 return {{}, whatwg_error::truncated_sequence, true};
84 auto b2 = static_cast<unsigned char>(*current);
85 ++current;
86 if (b2 < 0xA1 || b2 > 0xFE)
87 return {{}, whatwg_error::invalid_byte, true};
88 int pointer = (b1 - 0xA1) * 94 + (b2 - 0xA1);
89 char32_t cp = tables::euc_jp_jis0212[pointer];
90 if (cp == 0)
91 return {{}, whatwg_error::invalid_byte, true};
92 return {cp, {}, false};
93 }
94
95 // JIS X 0208: lead 0xA1-0xFE followed by trail 0xA1-0xFE
96 if (lead >= 0xA1 && lead <= 0xFE) {
97 if (current == end)
98 return {{}, whatwg_error::truncated_sequence, true};
99 auto trail = static_cast<unsigned char>(*current);
100 ++current;
101 if (trail < 0xA1 || trail > 0xFE)
102 return {{}, whatwg_error::invalid_byte, true};
103 int pointer = (lead - 0xA1) * 94 + (trail - 0xA1);
104 char32_t cp = tables::shift_jis[pointer];
105 if (cp == 0)
106 return {{}, whatwg_error::invalid_byte, true};
107 return {cp, {}, false};
108 }
109
110 // Any other byte is invalid
111 return {{}, whatwg_error::invalid_byte, true};
112}

114constexpr euc_jp_encode_result euc_jp_encode_one(char32_t cp) {
115 // ASCII passthrough
116 if (cp < 0x80) {
117 euc_jp_encode_result r{};
118 r.bytes[0] = static_cast<unsigned char>(cp);
119 r.count = 1;
120 return r;
121 }
122
123 // WHATWG special cases
124 if (cp == 0x00A5) {
125 euc_jp_encode_result r{};
126 r.bytes[0] = 0x5C;
127 r.count = 1;
128 return r;
129 }
130 if (cp == 0x203E) {
131 euc_jp_encode_result r{};
132 r.bytes[0] = 0x7E;
133 r.count = 1;
134 return r;
135 }
136
137 // Half-width katakana (U+FF61-U+FF9F) -> SS2 prefix + byte
138 if (cp >= 0xFF61 && cp <= 0xFF9F) {
139 euc_jp_encode_result r{};
140 r.bytes[0] = 0x8E;
141 r.bytes[1] = static_cast<unsigned char>(0xA1 + (cp - 0xFF61));
142 r.count = 2;
143 return r;
144 }
145
146 // Search JIS X 0208 table (first 8836 entries of shift_jis = EUC-JP jis0208 range)
147 for (int i = 0; i < 8836; ++i) {
148 if (tables::shift_jis[i] == cp) {
149 euc_jp_encode_result r{};
150 r.bytes[0] = static_cast<unsigned char>((i / 94) + 0xA1);
151 r.bytes[1] = static_cast<unsigned char>((i % 94) + 0xA1);
152 r.count = 2;
153 return r;
154 }
155 }
156
157 // Search JIS X 0212 table
158 for (int i = 0; i < 8836; ++i) {
159 if (tables::euc_jp_jis0212[i] == cp) {
160 euc_jp_encode_result r{};
161 r.bytes[0] = 0x8F;
162 r.bytes[1] = static_cast<unsigned char>((i / 94) + 0xA1);
163 r.bytes[2] = static_cast<unsigned char>((i % 94) + 0xA1);
164 r.count = 3;
165 return r;
166 }
167 }
168
169 return {{}, 0, true};
170}
