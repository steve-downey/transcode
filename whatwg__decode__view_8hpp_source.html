20 class iterator {
21 using base_iter = std::ranges::iterator_t<R>;
22 using base_sent = std::ranges::sentinel_t<R>;
23
24 base_iter current_;
25 base_sent end_;
26 char32_t value_{};
27 bool done_{false};
28
29 void load() {
30 if (current_ == end_) {
31 done_ = true;
32 return;
33 }
34 value_ = decode_one();
35 }
36
37 char32_t decode_one() {
38 auto byte = static_cast<unsigned char>(*current_);
39 ++current_;
40 // All WHATWG codecs share the 7-bit ASCII base (U+0000–U+007F):
41 // bytes 0x00–0x7F map directly to their code point value.
42 if (byte < 0x80)
43 return static_cast<char32_t>(byte);
44 // TODO: multi-byte UTF-8 sequences (step 5/6) and WHATWG error
45 // replacement rules (step 7) are not yet implemented. Any byte
46 // with the high bit set incorrectly emits U+FFFD here.
47 return U'�';
48 }

52 using value_type = char32_t;
53 using difference_type = std::ptrdiff_t;
54 using reference = char32_t;

78template <codec C>
79struct whatwg_decode_closure {
80 template <legacy_byte_range R>
81 auto operator()(R&& r) const {
82 return whatwg_decode_view<C, std::views::all_t<R>>(std::views::all(std::forward<R>(r)));
83 }
84
85 template <legacy_byte_range R>
86 friend auto operator|(R&& r, const whatwg_decode_closure& self) {
87 return self(std::forward<R>(r));
88 }
89
90 // Explicit diagnostic for raw arrays (char[N], const char[N], etc.).
91 // Arrays decay to pointers silently, which would silently include the null
92 // terminator in the transcoded input. The static_assert fires with a stable
93 // message instead of a deep substitution failure.
94 template <typename R>
95 requires std::is_array_v<std::remove_cvref_t<R>>
96 friend auto operator|(R&&, const whatwg_decode_closure&) {
97 static_assert(!std::is_array_v<std::remove_cvref_t<R>>,
98 "transcode: raw arrays are not valid input to whatwg_decode; "
99 "use beman::transcoding::views::null_term to get a range from a "
100 "null-terminated string");
101 }
102};
