70 // Explicit diagnostic for raw arrays (char[N], const char[N], etc.).
71 // Arrays decay to pointers silently, which would silently include the null
72 // terminator in the transcoded input. The static_assert fires with a stable
73 // message instead of a deep substitution failure.
74 template <typename R>
75 requires std::is_array_v<std::remove_cvref_t<R>>
76 friend auto operator|(R&&, const whatwg_decode_closure&) {
77 static_assert(!std::is_array_v<std::remove_cvref_t<R>>,
78 "transcode: raw arrays are not valid input to whatwg_decode; "
79 "use beman::transcoding::views::null_term to get a range from a "
80 "null-terminated string");
81 }
82};

178 requires legacy_byte_range<R>
179constexpr void whatwg_decode_view<C, R>::iterator::load() {
180 if (current_ == end_) {
181 done_ = true;
182 return;
183 }
184 if constexpr (C == codec::replacement) {
185 while (current_ != end_)
186 ++current_;
187 value_ = U'\xFFFD';
188 } else if constexpr (C == codec::utf_8) {
189 auto r = detail::utf8_decode_one(current_, end_);
190 value_ = r.is_error ? U'\xFFFD' : r.code_point;
191 }
