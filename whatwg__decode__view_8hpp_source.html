15template <codec C, std::ranges::input_range R>

17class whatwg_decode_view : public std::ranges::view_interface<whatwg_decode_view<C, R>> {
18 R base_;

20 class iterator {
21 using base_iter = std::ranges::iterator_t<R>;
22 using base_sent = std::ranges::sentinel_t<R>;
23
24 base_iter current_;
25 base_sent end_;
26 char32_t value_{};
27 bool done_{false};
28
29 constexpr void load() {
30 if (current_ == end_) {
31 done_ = true;
32 return;
33 }
34 value_ = decode_one();
35 }
36
37 constexpr char32_t decode_one() {
38 auto byte = static_cast<unsigned char>(*current_);
39 ++current_;
40
41 // ASCII fast path: all WHATWG codecs share the 7-bit base.
42 if (byte < 0x80)
43 return static_cast<char32_t>(byte);
44
45 // Determine sequence length and initial code point bits.
46 int extra;
47 char32_t cp;
48 if (byte >= 0xC2 && byte <= 0xDF) {
49 extra = 1;
50 cp = byte & 0x1F;
51 } else if (byte >= 0xE0 && byte <= 0xEF) {
52 extra = 2;
53 cp = byte & 0x0F;
54 } else if (byte >= 0xF0 && byte <= 0xF4) {
55 extra = 3;
56 cp = byte & 0x07;
57 } else {
58 // 0x80–0xBF: unexpected continuation byte.
59 // 0xC0–0xC1: always-overlong leads (pre-rejected).
60 // 0xF5–0xFF: out-of-range leads.
61 return U'�';
62 }
63
64 // Consume continuation bytes (0x80–0xBF each).
65 for (int i = 0; i < extra; ++i) {
66 if (current_ == end_)
67 return U'�'; // truncated sequence
68 auto cont = static_cast<unsigned char>(*current_);
69 if ((cont & 0xC0) != 0x80)
70 return U'�'; // bad continuation; do NOT advance —
71 // next load() re-processes this byte
72 ++current_;
73 cp = (cp << 6) | (cont & 0x3F);
74 }
75
76 // Validity: surrogates and out-of-Unicode-range.
77 if (cp >= 0xD800 && cp <= 0xDFFF)
78 return U'�';
79 if (cp > 0x10FFFF)
80 return U'�';
81
82 // Overlong check (should not be reachable given lead byte range
83 // above, but guard explicitly for clarity).
84 if (extra == 1 && cp < 0x80)
85 return U'�';
86 if (extra == 2 && cp < 0x800)
87 return U'�';
88 if (extra == 3 && cp < 0x10000)
89 return U'�';
90
91 return cp;
92 }

124template <codec C>
125struct whatwg_decode_closure {
126 template <legacy_byte_range R>
127 constexpr auto operator()(R&& r) const {
128 return whatwg_decode_view<C, std::views::all_t<R>>(std::views::all(std::forward<R>(r)));
129 }
130
131 template <legacy_byte_range R>
132 constexpr friend auto operator|(R&& r, const whatwg_decode_closure& self) {
133 return self(std::forward<R>(r));
134 }
135
136 // Explicit diagnostic for raw arrays (char[N], const char[N], etc.).
137 // Arrays decay to pointers silently, which would silently include the null
138 // terminator in the transcoded input. The static_assert fires with a stable
139 // message instead of a deep substitution failure.
140 template <typename R>
141 requires std::is_array_v<std::remove_cvref_t<R>>
142 friend auto operator|(R&&, const whatwg_decode_closure&) {
143 static_assert(!std::is_array_v<std::remove_cvref_t<R>>,
144 "transcode: raw arrays are not valid input to whatwg_decode; "
145 "use beman::transcoding::views::null_term to get a range from a "
146 "null-terminated string");
147 }
148};
