76struct whatwg_decode_closure {

78 constexpr auto operator()(R&& r) const {

87 // Explicit diagnostic for raw arrays (char[N], const char[N], etc.).
88 // Arrays decay to pointers silently, which would silently include the null
89 // terminator in the transcoded input. The static_assert fires with a stable
90 // message instead of a deep substitution failure.
91 template <typename R>
92 requires std::is_array_v<std::remove_cvref_t<R>>
93 friend auto operator|(R&&, const whatwg_decode_closure&) {
94 static_assert(!std::is_array_v<std::remove_cvref_t<R>>,
95 "transcode: raw arrays are not valid input to whatwg_decode; "
96 "use beman::transcoding::views::null_term to get a range from a "
97 "null-terminated string");
98 }
99};
