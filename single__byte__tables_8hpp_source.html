138constexpr char32_t decode_single_byte_at(unsigned char byte, const char32_t* table) {
139 if (byte < 0x80)
140 return static_cast<char32_t>(byte);
141 if (table == nullptr)
142 return static_cast<char32_t>(byte);
143 char32_t cp = table[byte - 0x80];
144 if (cp == 0)
145 return U'\xFFFD';
146 return cp;
147}
148
149constexpr char encode_single_byte_at(char32_t cp, const char32_t* table) {
150 if (cp < 0x80)
151 return static_cast<char>(cp);
152 if (table == nullptr)
153 return '?';
154 for (int i = 0; i < 128; ++i) {
155 if (table[i] == cp)
156 return static_cast<char>(0x80 + i);
157 }
158 return '?';
159}
