16// Decode bytes from `from` codec and re-encode to `to` codec.
17// Decode errors yield U+FFFD; unmapped encode codepoints yield '?'.
18// Returns empty string when `to` has no encoder (replacement, x_user_defined).
19std::string transcode_string(std::span<const char> src, codec from, codec to);

21// Label-based overload: looks up WHATWG labels for `from_label` and `to_label`.
22// Returns nullopt if either label is unknown.

46template <codec C>
47std::u32string transcode_decode_all(std::span<const char> src) {
48 std::u32string result;
49 for (char32_t cp : src | whatwg_decode<C>)
50 result.push_back(cp);
51 return result;
52}
53
54template <codec C>
55std::string transcode_encode_all(std::u32string_view src) {
56 std::string result;
57 for (char b : src | whatwg_encode<C>)
58 result.push_back(b);
59 return result;
60}
61
62} // namespace detail

317inline std::optional<std::string>
318transcode_string(std::span<const char> src, std::string_view from_label, std::string_view to_label) {
319 auto from = get_encoding(from_label);
320 auto to = get_encoding(to_label);
321 if (!from || !to)
322 return std::nullopt;
323 return transcode_string(src, *from, *to);
324}
