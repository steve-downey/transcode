31template <std::array<char32_t, 128> Table>
32template <std::input_iterator I, std::sentinel_for<I> S>
33constexpr decode_result table_codec<Table>::decode_one(I& current, [[maybe_unused]] S end) const {
34 auto byte = static_cast<unsigned char>(*current);
35 ++current;
36 if (byte < 0x80)
37 return {static_cast<char32_t>(byte), {}, false};
38 char32_t cp = Table[byte - 0x80];
39 if (cp == 0)
40 return {{}, decode_error::invalid_byte, true};
41 return {cp, {}, false};
42}

44template <std::array<char32_t, 128> Table>
45constexpr char32_t table_codec<Table>::decode_byte(unsigned char byte) const {
46 if (byte < 0x80)
47 return static_cast<char32_t>(byte);
48 char32_t cp = Table[byte - 0x80];
49 return cp == 0 ? U'\xFFFD' : cp;
50}

52template <std::array<char32_t, 128> Table>
53constexpr encode_result table_codec<Table>::encode_one(char32_t cp) const {
54 if (cp < 0x80)
55 return {{static_cast<unsigned char>(cp)}, 1, false};
56 for (int i = 0; i < 128; ++i) {
57 if (Table[static_cast<std::size_t>(i)] == cp)
58 return {{static_cast<unsigned char>(0x80 + i)}, 1, false};
59 }
60 return {{}, 0, true};
61}
