249template <encode_codec Codec, std::ranges::input_range R>
250 requires unicode_scalar_range<R>
251constexpr void encode_view<Codec, R>::iterator::load() {
252 if (current_ == end_) {
253 done_ = true;
254 return;
255 }
256 auto r = codec_.encode_one(*current_++);
257 if (r.is_error) {
258 buf_[0] = static_cast<unsigned char>('?');
259 len_ = 1;
260 } else {
261 for (int i = 0; i < r.count; ++i)
262 buf_[static_cast<std::size_t>(i)] = r.bytes[static_cast<std::size_t>(i)];
263 len_ = r.count;
264 }
265 pos_ = 0;
266}
