26// iconv_transcode_view<IconvFns, R>
27//
28// Adapts an input range R of legacy bytes into a byte-by-byte output range
29// in the target encoding, using an injected iconv-compatible callable set.
30// The iterator is intentionally move-only: iconv_t is an OS-managed opaque
31// handle that cannot be copied or shared.
32//
33// The caller provides an external output buffer (std::span<char>) to avoid
34// internal heap allocation.
35template <typename IconvFns, std::ranges::input_range R>
36 requires legacy_byte_range<R>
37class iconv_transcode_view : public std::ranges::view_interface<iconv_transcode_view<IconvFns, R>> {
38 R base_;
39 IconvFns fns_;
40 const char* from_;
41 const char* to_;
42 std::span<char> buffer_;
43
44 public:
45 class iterator {
46 using base_iter = std::ranges::iterator_t<R>;
47 using base_sent = std::ranges::sentinel_t<R>;
48
49 iconv_t handle_;
50 IconvFns fns_;
51 std::span<char> buffer_;
52 char* output_pos_;
53 char* output_end_;
54 // Accumulates unconsumed input bytes across load() calls so that
55 // multi-byte sequences can be assembled before passing to iconv.
56 char staging_[4];
57 size_t staging_len_;
58 base_iter current_;
59 base_sent end_;
60 bool done_;
61 bool flushed_;
62
63 iterator(const iterator&) = delete;
64 iterator& operator=(const iterator&) = delete;
65
66 // Fills output_pos_/output_end_ with the next batch of converted bytes.
67 // Handles EINVAL (incomplete sequence) by accumulating more input, and
68 // E2BIG/EILSEQ by yielding partial output or skipping one staging byte.
69 // Sets done_ = true when all input and staging bytes are exhausted.
70 void load();
71
72 // Only iconv_transcode_view::begin() may construct an iterator.
73 friend class iconv_transcode_view;
74 iterator(iconv_t handle, IconvFns fns, std::span<char> buffer, base_iter current, base_sent end);
75
76 public:
77 using iterator_concept = std::input_iterator_tag;
78 using value_type = char;
79 using difference_type = std::ptrdiff_t;
80 using reference = char;
81
82 iterator(iterator&&) noexcept;
83 iterator& operator=(iterator&&) noexcept;
84 ~iterator();
85
86 const base_iter& base() const noexcept { return current_; }
87
88 char operator*() const;
89 iterator& operator++();
90 void operator++(int);
91
92 friend bool operator==(const iterator& it, std::default_sentinel_t) { return it.done_; }
93 };
94
95 public:
96 explicit iconv_transcode_view(R base, IconvFns fns, const char* from, const char* to, std::span<char> buf);
97
98 const R& base() const& noexcept { return base_; }
99 R base() && { return std::move(base_); }
100
101 iterator begin();
102 std::default_sentinel_t end() const;
103};

149template <typename IconvFns, std::ranges::input_range R>
150 requires legacy_byte_range<R>
151void iconv_transcode_view<IconvFns, R>::iterator::load() {
152 while (true) {
153 // Append one input byte to staging (if any remains).
154 if (current_ != end_) {
155 staging_[staging_len_++] = static_cast<char>(*current_);
156 ++current_;
157 } else if (staging_len_ == 0) {
158 if (!flushed_) {
159 flushed_ = true;
160 char* out_ptr = buffer_.data();
161 size_t outleft = buffer_.size();
162 fns_.convert(handle_, nullptr, nullptr, &out_ptr, &outleft);
163 output_pos_ = buffer_.data();
164 output_end_ = out_ptr;
165 if (output_pos_ < output_end_)
166 return;
167 }
168 done_ = true;
169 return;
170 }
171 // else: staging holds bytes from a previous EINVAL; try to convert as-is.
172
173 char* in_ptr = staging_;
174 size_t inleft = staging_len_;
175 char* out_ptr = buffer_.data();
176 size_t outleft = buffer_.size();
177 size_t rc = fns_.convert(handle_, &in_ptr, &inleft, &out_ptr, &outleft);
178
179 output_pos_ = buffer_.data();
180 output_end_ = out_ptr;
181
182 // Shift consumed bytes out of staging_.
183 size_t consumed = staging_len_ - inleft;
184 if (consumed > 0) {
185 for (size_t i = 0; i < inleft; ++i)
186 staging_[i] = staging_[consumed + i];
187 staging_len_ = inleft;
188 }
189
190 if (rc != (size_t)-1) {
191 if (output_pos_ < output_end_)
192 return;
193 if (staging_len_ == 0 && current_ == end_) {
194 done_ = true;
195 return;
196 }
197 continue;
198 }
199
200 if (errno == EINVAL) {
201 // Incomplete input sequence: need more bytes before converting.
202 if (current_ == end_) {
203 // No more input; discard the partial sequence.
204 staging_len_ = 0;
205 done_ = true;
206 return;
207 }
208 // Loop to read another byte into staging.
209 if (output_pos_ < output_end_)
210 return;
211 continue;
212 }
213
214 // E2BIG or EILSEQ: yield whatever was written to the output buffer.
215 if (output_pos_ < output_end_)
216 return;
217 // Nothing written; skip one staging byte to avoid stalling.
218 if (staging_len_ > 0) {
219 for (size_t i = 0; i + 1 < staging_len_; ++i)
220 staging_[i] = staging_[i + 1];
221 --staging_len_;
222 }
223 if (staging_len_ == 0 && current_ == end_) {
224 done_ = true;
225 return;
226 }
227 }
228}

250template <typename IconvFns, std::ranges::input_range R>
251 requires legacy_byte_range<R>
252auto iconv_transcode_view<IconvFns, R>::iterator::operator=(iterator&& other) noexcept -> iterator& {
253 if (this != &other) {
254 if (handle_ != (iconv_t)-1)
255 fns_.close(handle_);
256 handle_ = other.handle_;
257 fns_ = std::move(other.fns_);
258 buffer_ = other.buffer_;
259 output_pos_ = other.output_pos_;
260 output_end_ = other.output_end_;
261 staging_len_ = other.staging_len_;
262 for (size_t i = 0; i < other.staging_len_; ++i)
263 staging_[i] = other.staging_[i];
264 current_ = std::move(other.current_);
265 end_ = std::move(other.end_);
266 done_ = other.done_;
267 flushed_ = other.flushed_;
268 other.handle_ = (iconv_t)-1;
269 other.done_ = true;
270 other.staging_len_ = 0;
271 }
272 return *this;
273}

303// iconv_transcode_closure<IconvFns> — pipe adapter for iconv_transcode_view.
304//
305// Stores the callable set, encoding pair, and output buffer so that
306// operator| can construct the view lazily:
307// auto v = input | iconv_transcode_closure<iconv_functions>{fns, "UTF-8", "UTF-32LE", buf};
308template <typename IconvFns>
309struct iconv_transcode_closure {
310 IconvFns fns_;
311 const char* from_;
312 const char* to_;
313 std::span<char> buffer_;
314
315 template <legacy_byte_range R>
316 auto operator()(R&& r) const {
317 return iconv_transcode_view<IconvFns, std::views::all_t<R>>(
318 std::views::all(std::forward<R>(r)), fns_, from_, to_, buffer_);
319 }
320
321 template <legacy_byte_range R>
322 friend auto operator|(R&& r, const iconv_transcode_closure& self) {
323 return self(std::forward<R>(r));
324 }
325};
