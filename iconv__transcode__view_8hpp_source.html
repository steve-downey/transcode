35// iconv_transcode_view<IconvFns, R>
36//
37// Adapts an input range R of legacy bytes into a byte-by-byte output range
38// in the target encoding, using an injected iconv-compatible callable set.
39// The iterator is intentionally move-only: iconv_t is an OS-managed opaque
40// handle that cannot be copied or shared.
41//
42// The caller provides an external output buffer (std::span<char>) to avoid
43// internal heap allocation.
44template <typename IconvFns, std::ranges::input_range R>
45 requires legacy_byte_range<R>
46class iconv_transcode_view : public std::ranges::view_interface<iconv_transcode_view<IconvFns, R>> {
47 R base_;
48 IconvFns fns_;
49 const char* from_;
50 const char* to_;
51 std::span<char> buffer_;
52
53 public:
54 class iterator {
55 using base_iter = std::ranges::iterator_t<R>;
56 using base_sent = std::ranges::sentinel_t<R>;
57
58 iconv_t handle_;
59 IconvFns fns_;
60 std::span<char> buffer_;
61 char* output_pos_;
62 char* output_end_;
63 // Accumulates unconsumed input bytes across load() calls so that
64 // multi-byte sequences can be assembled before passing to iconv.
65 char staging_[64];
66 size_t staging_len_{0};
67 base_iter current_;
68 base_sent end_;
69 bool done_;
70 bool flushed_{false};
71
72 // Fills output_pos_/output_end_ with the next batch of converted bytes.
73 // Handles EINVAL (incomplete sequence) by accumulating more input, and
74 // E2BIG/EILSEQ by yielding partial output or skipping one staging byte.
75 // Sets done_ = true when all input and staging bytes are exhausted.
76 void load();
77
78 // Only iconv_transcode_view::begin() may construct an iterator.
79 friend class iconv_transcode_view;
80 iterator(iconv_t handle, IconvFns fns, std::span<char> buffer, base_iter current, base_sent end);
81
82 public:
83 using iterator_concept = std::input_iterator_tag;
84 using value_type = char;
85 using difference_type = std::ptrdiff_t;
86 using reference = char;
87
88 iterator(const iterator&) = delete;
89 iterator& operator=(const iterator&) = delete;
90
91 iterator(iterator&&) noexcept;
92 iterator& operator=(iterator&&) noexcept;
93 ~iterator();
94
95 const base_iter& base() const noexcept { return current_; }
96
97 char operator*() const;
98 iterator& operator++();
99 void operator++(int);
100
101 friend bool operator==(const iterator& it, std::default_sentinel_t) { return it.done_; }
102 };

104 explicit iconv_transcode_view(R base, IconvFns fns, const char* from, const char* to, std::span<char> buf);

106 const R& base() const& noexcept { return base_; }
107 R base() && { return std::move(base_); }

109 iterator begin();
110 std::default_sentinel_t end() const;
111};

156template <typename IconvFns, std::ranges::input_range R>
157 requires legacy_byte_range<R>
158void iconv_transcode_view<IconvFns, R>::iterator::load() {
159 char* out_ptr = buffer_.data();
160 size_t outleft = buffer_.size();
161
162 while (outleft > 0) {
163 if constexpr (std::contiguous_iterator<base_iter> && std::sized_sentinel_for<base_sent, base_iter>) {
164 if (staging_len_ == 0 && current_ != end_) {
165 const auto* raw_ptr = reinterpret_cast<const char*>(std::to_address(current_));
166 auto remaining = static_cast<size_t>(end_ - current_);
167 char* in_ptr = const_cast<char*>(raw_ptr);
168 size_t inleft = remaining;
169 size_t rc = fns_.convert(handle_, &in_ptr, &inleft, &out_ptr, &outleft);
170 size_t consumed = remaining - inleft;
171 current_ += static_cast<std::ptrdiff_t>(consumed);
172 if (rc != iconv_error_rc) {
173 if (current_ == end_)
174 break;
175 continue;
176 }
177 if (errno == E2BIG)
178 break;
179 if (errno == EINVAL) {
180 if (inleft > 0 && inleft <= sizeof(staging_)) {
181 std::copy_n(in_ptr, inleft, staging_);
182 staging_len_ = inleft;
183 current_ += static_cast<std::ptrdiff_t>(inleft);
184 }
185 break;
186 }
187 // EILSEQ: skip one byte
188 if (current_ != end_)
189 ++current_;
190 if (current_ == end_ && staging_len_ == 0)
191 break;
192 continue;
193 }
194 }
195
196 // Slow path: fill staging from non-contiguous input or drain residue.
197 while (staging_len_ < sizeof(staging_) && current_ != end_) {
198 staging_[staging_len_++] = static_cast<char>(*current_);
199 ++current_;
200 }
201 if (staging_len_ == 0)
202 break;
203
204 char* in_ptr = staging_;
205 size_t inleft = staging_len_;
206 size_t rc = fns_.convert(handle_, &in_ptr, &inleft, &out_ptr, &outleft);
207 size_t consumed = staging_len_ - inleft;
208 if (consumed > 0 && inleft > 0)
209 std::copy_n(in_ptr, inleft, staging_);
210 staging_len_ = inleft;
211
212 if (rc != iconv_error_rc) {
213 if (staging_len_ == 0 && current_ == end_)
214 break;
215 continue;
216 }
217 if (errno == E2BIG)
218 break;
219 if (errno == EINVAL) {
220 if (current_ == end_) {
221 staging_len_ = 0;
222 break;
223 }
224 continue;
225 }
226 // EILSEQ: skip one staging byte.
227 if (staging_len_ > 0) {
228 std::copy_n(staging_ + 1, staging_len_ - 1, staging_);
229 --staging_len_;
230 }
231 if (staging_len_ == 0 && current_ == end_)
232 break;
233 }
234
235 output_pos_ = buffer_.data();
236 output_end_ = out_ptr;
237
238 if (output_pos_ == output_end_) {
239 if (!flushed_) {
240 flushed_ = true;
241 char* flush_out = buffer_.data();
242 size_t flush_left = buffer_.size();
243 fns_.convert(handle_, nullptr, nullptr, &flush_out, &flush_left);
244 output_pos_ = buffer_.data();
245 output_end_ = flush_out;
246 }
247 if (output_pos_ == output_end_)
248 done_ = true;
249 }
250}

252template <typename IconvFns, std::ranges::input_range R>

254iconv_transcode_view<IconvFns, R>::iterator::iterator(iterator&& other) noexcept

325// iconv_transcode_closure<IconvFns> — pipe adapter for iconv_transcode_view.
326//
327// Stores the callable set, encoding pair, and output buffer so that
328// operator| can construct the view lazily:
329// auto v = input | iconv_transcode_closure<iconv_functions>{fns, "UTF-8", "UTF-32LE", buf};
330template <typename IconvFns>
331struct iconv_transcode_closure {
332 IconvFns fns_;
333 const char* from_;
334 const char* to_;
335 std::span<char> buffer_;
336
337 template <legacy_byte_range R>
338 auto operator()(R&& r) const {
339 return iconv_transcode_view<IconvFns, std::views::all_t<R>>(
340 std::views::all(std::forward<R>(r)), fns_, from_, to_, buffer_);
341 }
342
343 template <legacy_byte_range R>
344 friend auto operator|(R&& r, const iconv_transcode_closure& self) {
345 return self(std::forward<R>(r));
346 }
347};
