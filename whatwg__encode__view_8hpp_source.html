206template <codec C, std::ranges::input_range R>
207 requires unicode_scalar_range<R>
208constexpr void whatwg_encode_view<C, R>::iterator::load() {
209 if (current_ == end_) {
210 done_ = true;
211 return;
212 }
213 auto encode_single = [&](const char32_t (&table)[128]) {
214 auto r = detail::single_byte_encode_one(current_, end_, table);
215 buf_[0] = r.is_error ? '?' : static_cast<char>(r.byte);
216 len_ = 1;
217 pos_ = 0;
218 };
219 if constexpr (C == codec::utf_8) {
220 auto r = detail::utf8_encode_one(static_cast<char32_t>(*current_));
221 ++current_;
222 if (r.is_error) {
223 buf_[0] = '\xEF';
224 buf_[1] = '\xBF';
225 buf_[2] = '\xBD';
226 len_ = 3;
227 } else {
228 for (int i = 0; i < r.count; ++i)
229 buf_[i] = r.bytes[i];
230 len_ = r.count;
231 }
232 pos_ = 0;
233 } else if constexpr (C == codec::ibm866) {
234 encode_single(detail::tables::ibm866);
235 } else if constexpr (C == codec::iso_8859_2) {
236 encode_single(detail::tables::iso_8859_2);
237 } else if constexpr (C == codec::iso_8859_3) {
238 encode_single(detail::tables::iso_8859_3);
239 } else if constexpr (C == codec::iso_8859_4) {
240 encode_single(detail::tables::iso_8859_4);
241 } else if constexpr (C == codec::iso_8859_5) {
242 encode_single(detail::tables::iso_8859_5);
243 } else if constexpr (C == codec::iso_8859_6) {
244 encode_single(detail::tables::iso_8859_6);
245 } else if constexpr (C == codec::iso_8859_7) {
246 encode_single(detail::tables::iso_8859_7);
247 } else if constexpr (C == codec::iso_8859_8 || C == codec::iso_8859_8_i) {
248 encode_single(detail::tables::iso_8859_8);
249 } else if constexpr (C == codec::iso_8859_10) {
250 encode_single(detail::tables::iso_8859_10);
251 } else if constexpr (C == codec::iso_8859_13) {
252 encode_single(detail::tables::iso_8859_13);
253 } else if constexpr (C == codec::iso_8859_14) {
254 encode_single(detail::tables::iso_8859_14);
255 } else if constexpr (C == codec::iso_8859_15) {
256 encode_single(detail::tables::iso_8859_15);
257 } else if constexpr (C == codec::iso_8859_16) {
258 encode_single(detail::tables::iso_8859_16);
259 } else if constexpr (C == codec::koi8_r) {
260 encode_single(detail::tables::koi8_r);
261 } else if constexpr (C == codec::koi8_u) {
262 encode_single(detail::tables::koi8_u);
263 } else if constexpr (C == codec::macintosh) {
264 encode_single(detail::tables::macintosh);
265 } else if constexpr (C == codec::windows_874) {
266 encode_single(detail::tables::windows_874);
267 } else if constexpr (C == codec::windows_1250) {
268 encode_single(detail::tables::windows_1250);
269 } else if constexpr (C == codec::windows_1251) {
270 encode_single(detail::tables::windows_1251);
271 } else if constexpr (C == codec::windows_1252) {
272 encode_single(detail::tables::windows_1252);
273 } else if constexpr (C == codec::windows_1253) {
274 encode_single(detail::tables::windows_1253);
275 } else if constexpr (C == codec::windows_1254) {
276 encode_single(detail::tables::windows_1254);
277 } else if constexpr (C == codec::windows_1255) {
278 encode_single(detail::tables::windows_1255);
279 } else if constexpr (C == codec::windows_1256) {
280 encode_single(detail::tables::windows_1256);
281 } else if constexpr (C == codec::windows_1257) {
282 encode_single(detail::tables::windows_1257);
283 } else if constexpr (C == codec::windows_1258) {
284 encode_single(detail::tables::windows_1258);
285 } else if constexpr (C == codec::x_mac_cyrillic) {
286 encode_single(detail::tables::x_mac_cyrillic);
287 } else if constexpr (C == codec::utf_16be) {
288 auto r = detail::utf16be_encode_one(static_cast<char32_t>(*current_));
289 ++current_;
290 if (r.is_error) {
291 // U+FFFD in UTF-16BE: 0xFF 0xFD
292 buf_[0] = '\xFF';
293 buf_[1] = '\xFD';
294 len_ = 2;
295 } else {
296 for (int i = 0; i < r.count; ++i)
297 buf_[i] = r.bytes[i];
298 len_ = r.count;
299 }
300 pos_ = 0;
301 } else if constexpr (C == codec::utf_16le) {
302 auto r = detail::utf16le_encode_one(static_cast<char32_t>(*current_));
303 ++current_;
304 if (r.is_error) {
305 // U+FFFD in UTF-16LE: 0xFD 0xFF
306 buf_[0] = '\xFD';
307 buf_[1] = '\xFF';
308 len_ = 2;
309 } else {
310 for (int i = 0; i < r.count; ++i)
311 buf_[i] = r.bytes[i];
312 len_ = r.count;
313 }
314 pos_ = 0;
315 } else if constexpr (C == codec::gbk) {
316 auto r = detail::gbk_encode_one(static_cast<char32_t>(*current_));
317 ++current_;
318 if (r.is_error) {
319 buf_[0] = '?';
320 len_ = 1;
321 } else {
322 for (int i = 0; i < r.count; ++i)
323 buf_[i] = static_cast<char>(r.bytes[i]);
324 len_ = r.count;
325 }
326 pos_ = 0;
327 }
